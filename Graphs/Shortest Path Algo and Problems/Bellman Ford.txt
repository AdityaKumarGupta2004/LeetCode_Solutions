
class Solution {
    public int[] bellmanFord(int V, int[][] edges, int S) {
        // code here
        int[] dist = new int[V];
        for (int i = 0; i < V; i++) dist[i] = (int)(1e8);
        dist[S] = 0;
        // V x E
        for (int i = 0; i < V - 1; i++) {
            for (int[] it : edges) {
                int u = it[0]; //Starting point of the edge
                int v = it[1];//Ending point of the edge
                int wt = it[2]; //Edge weight
                if (dist[u] != 1e8 && dist[u] + wt < dist[v]) {
                    dist[v] = dist[u] + wt;
                }
            }
        }
        // Nth relaxation to check negative cycle
        for (int[] it : edges) {
            int u = it[0]; //Starting point of the edge
                int v = it[1];//Ending point of the edge
                int wt = it[2]; //Edge weight
            if (dist[u] != 1e8 && dist[u] + wt < dist[v]) {
                int temp[] = new int[1];
                temp[0] = -1;
                return temp;
            }
        }
        return dist;
    }
}


// n-1 iteration is done here 
Tc = O(V x E)
